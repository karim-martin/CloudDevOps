CI/CD Pipeline

Set up GitHub Actions CI/CD pipeline using Workload Identity Federation for secure, keyless authentication to GCP.

Step 1: Configure Workload Identity Federation

Allow GitHub Actions to authenticate to GCP without service account keys.

# Set variables
export PROJECT_ID="your-project-id"
export PROJECT_NUMBER=$(gcloud projects describe $PROJECT_ID --format='value(projectNumber)')
export GITHUB_ORG="your-github-org"      # or username
export GITHUB_REPO="your-repo-name"
export POOL_NAME="github-pool"
export PROVIDER_NAME="github-provider"
export SA_NAME="github-actions-sa"

# Create Workload Identity Pool
gcloud iam workload-identity-pools create $POOL_NAME \
    --location="global" \
    --display-name="GitHub Actions Pool"

# Create Workload Identity Provider
gcloud iam workload-identity-pools providers create-oidc $PROVIDER_NAME \
    --location="global" \
    --workload-identity-pool=$POOL_NAME \
    --display-name="GitHub Provider" \
    --attribute-mapping="google.subject=assertion.sub,attribute.actor=assertion.actor,attribute.repository=assertion.repository,attribute.repository_owner=assertion.repository_owner" \
    --issuer-uri="https://token.actions.githubusercontent.com"

# Get the full provider name
export WORKLOAD_IDENTITY_PROVIDER="projects/$PROJECT_NUMBER/locations/global/workloadIdentityPools/$POOL_NAME/providers/$PROVIDER_NAME"
echo "Workload Identity Provider: $WORKLOAD_IDENTITY_PROVIDER"

Step 2: Create CI/CD Service Account

# Create service account for GitHub Actions
gcloud iam service-accounts create $SA_NAME \
    --display-name="GitHub Actions Service Account"

# Grant required permissions
# Artifact Registry (push images)
gcloud projects add-iam-policy-binding $PROJECT_ID \
    --member="serviceAccount:$SA_NAME@$PROJECT_ID.iam.gserviceaccount.com" \
    --role="roles/artifactregistry.writer"

# GKE (deploy to cluster)
gcloud projects add-iam-policy-binding $PROJECT_ID \
    --member="serviceAccount:$SA_NAME@$PROJECT_ID.iam.gserviceaccount.com" \
    --role="roles/container.developer"

# Allow GitHub to impersonate the service account
gcloud iam service-accounts add-iam-policy-binding \
    $SA_NAME@$PROJECT_ID.iam.gserviceaccount.com \
    --role="roles/iam.workloadIdentityUser" \
    --member="principalSet://iam.googleapis.com/projects/$PROJECT_NUMBER/locations/global/workloadIdentityPools/$POOL_NAME/attribute.repository/$GITHUB_ORG/$GITHUB_REPO"

Step 3: Add GitHub Secrets

Add these secrets to your GitHub repository (Settings → Secrets → Actions):

Step 4: GitHub Actions Workflow

# .github/workflows/deploy.yaml
name: Build and Deploy to GKE

on:
  push:
    branches:
      - main
  pull_request:
    branches:
      - main

env:
  PROJECT_ID: ${{ secrets.GCP_PROJECT_ID }}
  GKE_CLUSTER: ${{ secrets.GKE_CLUSTER }}
  GKE_REGION: ${{ secrets.GKE_REGION }}
  REGISTRY: us-central1-docker.pkg.dev
  REPOSITORY: app-repo

jobs:
  build-and-deploy:
    name: Build and Deploy
    runs-on: ubuntu-latest

    permissions:
      contents: read
      id-token: write  # Required for Workload Identity Federation

    steps:
    - name: Checkout code
      uses: actions/checkout@v4

    - name: Authenticate to Google Cloud
      id: auth
      uses: google-github-actions/auth@v2
      with:
        workload_identity_provider: ${{ secrets.GCP_WORKLOAD_IDENTITY_PROVIDER }}
        service_account: ${{ secrets.GCP_SERVICE_ACCOUNT }}

    - name: Set up Cloud SDK
      uses: google-github-actions/setup-gcloud@v2

    - name: Configure Docker for Artifact Registry
      run: |
        gcloud auth configure-docker ${{ env.REGISTRY }} --quiet

    - name: Build Frontend Image
      run: |
        docker build -t ${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPOSITORY }}/frontend:${{ github.sha }} \
                     -t ${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPOSITORY }}/frontend:latest \
                     -f frontend/Dockerfile frontend/

    - name: Build Backend Image
      run: |
        docker build -t ${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPOSITORY }}/backend:${{ github.sha }} \
                     -t ${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPOSITORY }}/backend:latest \
                     -f backend/Dockerfile backend/

    - name: Push Frontend Image
      run: |
        docker push ${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPOSITORY }}/frontend:${{ github.sha }}
        docker push ${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPOSITORY }}/frontend:latest

    - name: Push Backend Image
      run: |
        docker push ${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPOSITORY }}/backend:${{ github.sha }}
        docker push ${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPOSITORY }}/backend:latest

    - name: Get GKE credentials
      uses: google-github-actions/get-gke-credentials@v2
      with:
        cluster_name: ${{ env.GKE_CLUSTER }}
        location: ${{ env.GKE_REGION }}

    - name: Deploy to GKE
      run: |
        # Update image tags in deployment files
        sed -i "s|IMAGE_TAG|${{ github.sha }}|g" k8s/frontend/deployment.yaml
        sed -i "s|IMAGE_TAG|${{ github.sha }}|g" k8s/backend/deployment.yaml
        sed -i "s|PROJECT_ID|${{ env.PROJECT_ID }}|g" k8s/frontend/deployment.yaml
        sed -i "s|PROJECT_ID|${{ env.PROJECT_ID }}|g" k8s/backend/deployment.yaml

        # Apply Kubernetes manifests
        kubectl apply -f k8s/frontend/
        kubectl apply -f k8s/backend/

        # Wait for rollout
        kubectl rollout status deployment/react-frontend -n app --timeout=300s
        kubectl rollout status deployment/dotnet-api -n app --timeout=300s

    - name: Verify deployment
      run: |
        kubectl get pods -n app
        kubectl get services -n app

Pull Request Workflow (Build Only)

# .github/workflows/pr.yaml
name: PR Build Check

on:
  pull_request:
    branches:
      - main

env:
  PROJECT_ID: ${{ secrets.GCP_PROJECT_ID }}
  REGISTRY: us-central1-docker.pkg.dev
  REPOSITORY: app-repo

jobs:
  build:
    name: Build Images
    runs-on: ubuntu-latest

    permissions:
      contents: read
      id-token: write

    steps:
    - name: Checkout code
      uses: actions/checkout@v4

    - name: Authenticate to Google Cloud
      uses: google-github-actions/auth@v2

        workload_identity_provider: ${{ secrets.GCP_WORKLOAD_IDENTITY_PROVIDER }}
        service_account: ${{ secrets.GCP_SERVICE_ACCOUNT }}

    - name: Configure Docker for Artifact Registry

                     -f backend/Dockerfile backend/

    - name: Run Tests
      run: |
        # Add your test commands here
        echo "Running tests..."
        # cd frontend && npm test
        # cd backend && dotnet test

Deployment YAML with Image Tag

# k8s/backend/deployment.yaml (template)
apiVersion: apps/v1
kind: Deployment
metadata:
  name: dotnet-api
  namespace: app
spec:
  replicas: 3
  selector:
    matchLabels:
      app: dotnet-api
  template:
    metadata:
      labels:
        app: dotnet-api
    spec:
      serviceAccountName: app-ksa
      containers:
      - name: api
        # IMAGE_TAG and PROJECT_ID replaced by CI/CD pipeline
        image: us-central1-docker.pkg.dev/PROJECT_ID/app-repo/backend:IMAGE_TAG
        ports:
        - containerPort: 8080

Rollback Procedure

# View rollout history
kubectl rollout history deployment/dotnet-api -n app

# Rollback to previous version
kubectl rollout undo deployment/dotnet-api -n app

# Rollback to specific revision
kubectl rollout undo deployment/dotnet-api -n app --to-revision=2

# Check rollback status
kubectl rollout status deployment/dotnet-api -n app

# If needed, manually update to specific image
kubectl set image deployment/dotnet-api \
    api=us-central1-docker.pkg.dev/PROJECT_ID/app-repo/backend:abc123 \
    -n app

Branch-Based Environments (Optional)

# Deploy to different namespaces based on branch
- name: Set environment
  run: |
    if [[ "${{ github.ref }}" == "refs/heads/main" ]]; then
      echo "NAMESPACE=production" >> $GITHUB_ENV
    elif [[ "${{ github.ref }}" == "refs/heads/staging" ]]; then
      echo "NAMESPACE=staging" >> $GITHUB_ENV
    else
      echo "NAMESPACE=development" >> $GITHUB_ENV
    fi

- name: Deploy
  run: |
    kubectl apply -f k8s/ -n ${{ env.NAMESPACE }}

Why Workload Identity Federation?

Workload Identity Federation eliminates the need for long-lived service account keys. GitHub's OIDC tokens are exchanged for short-lived GCP credentials, reducing security risks from key exposure.

id-token Permission

The permissions: id-token: write is required for Workload Identity Federation. Without it, GitHub Actions cannot request the OIDC token needed for GCP authentication.

Image Tags

Using ${{ github.sha }} as the image tag ensures each deployment uses a unique, traceable image. The latest tag is also pushed for convenience but should not be used in production deployments.
