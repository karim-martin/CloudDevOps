Deploy .NET Backend

Deploy the .NET 8 backend API to GKE with 3 replicas, database connection via Secret, and Workload Identity for GCP access.

Deployment

# k8s/backend/deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: dotnet-api
  namespace: app
  labels:
    app: dotnet-api
spec:
  replicas: 3
  selector:
    matchLabels:
      app: dotnet-api
  template:
    metadata:
      labels:
        app: dotnet-api
    spec:
      serviceAccountName: app-ksa
      containers:
      - name: api
        image: us-central1-docker.pkg.dev/PROJECT_ID/app-repo/backend:latest
        ports:
        - containerPort: 8080
          name: http
        env:
        - name: ASPNETCORE_ENVIRONMENT
          value: "Production"
        - name: ConnectionStrings__DefaultConnection
          valueFrom:
            secretKeyRef:
              name: db-credentials
              key: connection-string
        - name: GCS_BUCKET
          valueFrom:
            configMapKeyRef:
              name: backend-config
              key: gcs-bucket
        resources:
          requests:
            memory: "128Mi"
            cpu: "100m"
          limits:
            memory: "512Mi"
            cpu: "500m"
        readinessProbe:
          httpGet:
            path: /health/ready
            port: 8080
          initialDelaySeconds: 10
          periodSeconds: 10
          timeoutSeconds: 5
          failureThreshold: 3
        livenessProbe:
          httpGet:
            path: /health/live
            port: 8080
          initialDelaySeconds: 15
          periodSeconds: 30
          timeoutSeconds: 5
          failureThreshold: 3
        startupProbe:
          httpGet:
            path: /health
            port: 8080
          initialDelaySeconds: 5
          periodSeconds: 5
          failureThreshold: 30

Service

# k8s/backend/service.yaml
apiVersion: v1
kind: Service
metadata:
  name: dotnet-api
  namespace: app
  labels:
    app: dotnet-api
spec:
  type: ClusterIP
  selector:
    app: dotnet-api
  ports:
  - port: 8080
    targetPort: 8080
    name: http

ConfigMap

# k8s/backend/configmap.yaml
apiVersion: v1
kind: ConfigMap
metadata:
  name: backend-config
  namespace: app
data:
  gcs-bucket: "PROJECT_ID-app-uploads"
  log-level: "Information"
  allowed-origins: "https://app.example.com"

Secret (for reference - create via kubectl)

# k8s/backend/secret.yaml (DO NOT commit with real values)
apiVersion: v1
kind: Secret
metadata:
  name: db-credentials
  namespace: app
type: Opaque
stringData:
  connection-string: "Server=10.x.x.x;Database=appdb;User=appuser;Password=SECURE_PASSWORD;SslMode=Required;"
# Better: Create secret via kubectl (not committed to git)
kubectl create secret generic db-credentials \
    --namespace=app \
    --from-literal=connection-string="Server=${DB_HOST};Database=appdb;User=appuser;Password=${DB_PASSWORD};SslMode=Required;"

Deploy to Cluster

# Ensure namespace exists
kubectl create namespace app --dry-run=client -o yaml | kubectl apply -f -

# Create secret first (if not already created)
kubectl create secret generic db-credentials \
    --namespace=app \
    --from-literal=connection-string="Server=${DB_HOST};Database=appdb;User=appuser;Password=${DB_PASSWORD};SslMode=Required;"

# Apply ConfigMap
kubectl apply -f k8s/backend/configmap.yaml

# Apply Deployment
kubectl apply -f k8s/backend/deployment.yaml

# Apply Service
kubectl apply -f k8s/backend/service.yaml

# Or apply all non-secret files
kubectl apply -f k8s/backend/configmap.yaml -f k8s/backend/deployment.yaml -f k8s/backend/service.yaml

Verify Deployment

# Check deployment status
kubectl get deployment dotnet-api -n app

# Check pods are running
kubectl get pods -l app=dotnet-api -n app

# Check for any issues
kubectl describe pod -l app=dotnet-api -n app

# View logs
kubectl logs -l app=dotnet-api -n app --tail=100

# Check if database connection works
kubectl logs -l app=dotnet-api -n app | grep -i "database\|connection\|error"

# Watch rollout
kubectl rollout status deployment/dotnet-api -n app

Test Backend

# Port forward to test locally
kubectl port-forward service/dotnet-api 8080:8080 -n app

# Test endpoints
curl http://localhost:8080/health
curl http://localhost:8080/health/ready
curl http://localhost:8080/api/your-endpoint

# Test from inside the cluster
kubectl run test-curl --rm -it --restart=Never --image=curlimages/curl -- \
    curl http://dotnet-api.app.svc.cluster.local:8080/health

Verify Workload Identity

# Verify pod is using the correct service account
kubectl get pod -l app=dotnet-api -n app -o jsonpath='{.items[0].spec.serviceAccountName}'
# Should output: app-ksa

# Test GCS access from pod
kubectl exec -it deployment/dotnet-api -n app -- \
    curl -H "Metadata-Flavor: Google" \
    http://metadata.google.internal/computeMetadata/v1/instance/service-accounts/default/email
# Should output: app-workload-sa@PROJECT_ID.iam.gserviceaccount.com

Troubleshooting

IssueCommand
Pod not startingkubectl describe pod -l app=dotnet-api -n app
Image pull errorkubectl get events -n app --sort-by=.lastTimestamp
Database connectionkubectl logs -l app=dotnet-api -n app | grep -i error
Secret not foundkubectl get secret db-credentials -n app
Health check failingkubectl port-forward pod/POD_NAME 8080:8080 && curl localhost:8080/health

Update Secret

# Update database credentials
kubectl delete secret db-credentials -n app
kubectl create secret generic db-credentials \
    --namespace=app \
    --from-literal=connection-string="NEW_CONNECTION_STRING"

# Restart pods to pick up new secret
kubectl rollout restart deployment/dotnet-api -n app
Startup Probe

The startupProbe gives the app 150 seconds (30 failures Ã— 5 seconds) to start. This is important for .NET apps that may take time to initialize database connections.

Service Account

The deployment uses serviceAccountName: app-ksa which is linked to the GCP service account via Workload Identity. This allows the pod to access Cloud SQL and Cloud Storage without credential files.
