Deploy React Frontend

Deploy the React frontend to GKE with 3 replicas, ConfigMap for configuration, and a ClusterIP Service.

Deployment

# k8s/frontend/deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: react-frontend
  namespace: app
  labels:
    app: react-frontend
spec:
  replicas: 3
  selector:
    matchLabels:
      app: react-frontend
  template:
    metadata:
      labels:
        app: react-frontend
    spec:
      containers:
      - name: react
        image: us-central1-docker.pkg.dev/PROJECT_ID/app-repo/frontend:latest
        ports:
        - containerPort: 80
          name: http
        resources:
          requests:
            memory: "64Mi"
            cpu: "50m"
          limits:
            memory: "128Mi"
            cpu: "100m"
        readinessProbe:
          httpGet:
            path: /health
            port: 80
          initialDelaySeconds: 5
          periodSeconds: 10
          timeoutSeconds: 3
          failureThreshold: 3
        livenessProbe:
          httpGet:
            path: /health
            port: 80
          initialDelaySeconds: 10
          periodSeconds: 30
          timeoutSeconds: 3
          failureThreshold: 3
        volumeMounts:
        - name: nginx-config
          mountPath: /etc/nginx/conf.d/default.conf
          subPath: nginx.conf
      volumes:
      - name: nginx-config
        configMap:
          name: frontend-config

Service

# k8s/frontend/service.yaml
apiVersion: v1
kind: Service
metadata:
  name: react-frontend
  namespace: app
  labels:
    app: react-frontend
spec:
  type: ClusterIP
  selector:
    app: react-frontend
  ports:
  - port: 80
    targetPort: 80
    name: http

ConfigMap

# k8s/frontend/configmap.yaml
apiVersion: v1
kind: ConfigMap
metadata:
  name: frontend-config
  namespace: app
data:
  nginx.conf: |
    server {
        listen 80;
        server_name _;
        root /usr/share/nginx/html;
        index index.html;

        location / {
            try_files $uri $uri/ /index.html;
        }

        location /api/ {
            proxy_pass http://dotnet-api:8080/api/;
            proxy_http_version 1.1;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
        }

        location /health {
            access_log off;
            return 200 "healthy\n";
            add_header Content-Type text/plain;
        }
    }

Deploy to Cluster

# Ensure you're in the app namespace
kubectl config set-context --current --namespace=app

# Apply ConfigMap first
kubectl apply -f k8s/frontend/configmap.yaml

# Apply Deployment
kubectl apply -f k8s/frontend/deployment.yaml

# Apply Service
kubectl apply -f k8s/frontend/service.yaml

# Or apply all at once
kubectl apply -f k8s/frontend/

Verify Deployment

# Check deployment status
kubectl get deployment react-frontend -n app

# Check pods are running
kubectl get pods -l app=react-frontend -n app

# Check pod details
kubectl describe pod -l app=react-frontend -n app

# Check service
kubectl get service react-frontend -n app

# View logs
kubectl logs -l app=react-frontend -n app --tail=50

# Watch rollout status
kubectl rollout status deployment/react-frontend -n app

Test Frontend

# Port forward to test locally
kubectl port-forward service/react-frontend 3000:80 -n app

# In another terminal, test
curl http://localhost:3000
curl http://localhost:3000/health

# Test from inside the cluster
kubectl run test-curl --rm -it --restart=Never --image=curlimages/curl -- \
    curl http://react-frontend.app.svc.cluster.local/health

Update Deployment

# Update image (triggers rolling update)
kubectl set image deployment/react-frontend \
    react=us-central1-docker.pkg.dev/PROJECT_ID/app-repo/frontend:v2 \
    -n app

# Or update via YAML edit
kubectl edit deployment react-frontend -n app

# Check rollout status
kubectl rollout status deployment/react-frontend -n app

# Rollback if needed
kubectl rollout undo deployment/react-frontend -n app

# View rollout history
kubectl rollout history deployment/react-frontend -n app

Nginx Proxy

The nginx config proxies /api/* requests to the backend service (dotnet-api:8080). This allows the frontend to call APIs without CORS issues, as both appear to be from the same origin.

Replace PROJECT_ID

Remember to replace PROJECT_ID in the image path with your actual GCP project ID before applying the manifests.
