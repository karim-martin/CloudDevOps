Docker Compose

Docker Compose is a tool for defining and running multi-container applications.

Compose Commands

CommandDescription
docker compose upCreate and start containers
docker compose up -dStart in detached mode
docker compose up --buildBuild images before starting
docker compose downStop and remove containers
docker compose down -vAlso remove volumes
docker compose psList containers
docker compose logsView logs
docker compose logs -fFollow logs
docker compose exec service cmdExecute command in service
docker compose buildBuild images
docker compose pullPull images
docker compose restartRestart services

Example: Web App with Database

# docker-compose.yml
version: '3.8'

services:
  web:
    build: .
    ports:
      - "3000:3000"
    environment:
      - NODE_ENV=production
      - DATABASE_URL=postgres://user:pass@db:5432/mydb
    depends_on:
      - db
    volumes:
      - ./app:/app
    restart: unless-stopped

  db:
    image: postgres:15
    environment:
      - POSTGRES_USER=user
      - POSTGRES_PASSWORD=pass
      - POSTGRES_DB=mydb
    volumes:
      - postgres_data:/var/lib/postgresql/data
    ports:
      - "5432:5432"

  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"

volumes:
  postgres_data:

Common Service Options

services:
  app:
    # Build from Dockerfile
    build:
      context: .
      dockerfile: Dockerfile.prod
      args:
        - VERSION=1.0

    # Or use image directly
    image: nginx:latest

    # Container name
    container_name: my-app

    # Port mapping
    ports:
      - "8080:80"      # host:container
      - "443:443"

    # Environment variables
    environment:
      - NODE_ENV=production
      - API_KEY=${API_KEY}  # from .env file

    # Or from file
    env_file:
      - .env

    # Volume mounts
    volumes:
      - ./data:/app/data           # bind mount
      - app_data:/app/storage       # named volume
      - /app/node_modules           # anonymous volume

    # Dependencies
    depends_on:
      - db
      - redis

    # Restart policy
    restart: always  # no, on-failure, unless-stopped

    # Resource limits
    deploy:
      resources:
        limits:
          cpus: '0.5'
          memory: 512M

    # Health check
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost/health"]
      interval: 30s
      timeout: 10s
      retries: 3

    # Networks
    networks:
      - frontend
      - backend

networks:
  frontend:
  backend:

volumes:
  app_data:

Environment Files

# .env file
POSTGRES_USER=myuser
POSTGRES_PASSWORD=secret
API_KEY=abc123
# Reference in docker-compose.yml
services:
  db:
    image: postgres
    environment:
      - POSTGRES_USER=${POSTGRES_USER}
      - POSTGRES_PASSWORD=${POSTGRES_PASSWORD}

Profiles

Run specific services based on profiles:

services:
  web:
    image: nginx

  debug:
    image: busybox
    profiles:
      - debug
# Run with debug profile
docker compose --profile debug up
