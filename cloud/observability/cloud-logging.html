Cloud Logging

Cloud Logging collects, stores, and analyzes logs from GKE, Cloud Run, and all GCP services.

Log Query Language

# GKE container logs
resource.type="k8s_container"
resource.labels.namespace_name="production"
resource.labels.container_name="api"

# Filter by severity
resource.type="k8s_container"
severity>=ERROR

# Search text in logs
resource.type="k8s_container"
textPayload:"connection refused"

# JSON payload queries
resource.type="k8s_container"
jsonPayload.level="error"
jsonPayload.userId="12345"

# Time range
resource.type="k8s_container"
timestamp>="2024-01-01T00:00:00Z"
timestamp<="2024-01-02T00:00:00Z"

# Cloud Run logs
resource.type="cloud_run_revision"
resource.labels.service_name="my-service"

# Load balancer logs
resource.type="http_load_balancer"
httpRequest.status>=500

Structured Logging

// .NET - Structured logging (auto-parsed by Cloud Logging)
using Microsoft.Extensions.Logging;

logger.LogInformation("Processing order {OrderId} for user {UserId}",
    orderId, userId);

// Output JSON for Cloud Logging
// {"message": "Processing order", "orderId": "123", "userId": "456", "severity": "INFO"}
// Node.js - Structured logging
const {Logging} = require('@google-cloud/logging');
const logging = new Logging();
const log = logging.log('my-log');

const entry = log.entry(
  {resource: {type: 'global'}},
  {
    message: 'User logged in',
    userId: '12345',
    action: 'login'
  }
);
await log.write(entry);

Log-Based Metrics

# Create counter metric from logs
gcloud logging metrics create error-count \
    --description="Count of error logs" \
    --log-filter='resource.type="k8s_container" severity=ERROR'

# Create distribution metric
gcloud logging metrics create request-latency \
    --description="Request latency distribution" \
    --log-filter='resource.type="http_load_balancer"' \
    --bucket-boundaries=0.1,0.5,1,2,5,10

# List metrics
gcloud logging metrics list

Log Sinks (Export)

# Export to Cloud Storage
gcloud logging sinks create storage-sink \
    storage.googleapis.com/my-log-bucket \
    --log-filter='resource.type="k8s_container"'

# Export to BigQuery
gcloud logging sinks create bq-sink \
    bigquery.googleapis.com/projects/PROJECT/datasets/logs \
    --log-filter='resource.type="k8s_container" severity>=WARNING'

# Export to Pub/Sub
gcloud logging sinks create pubsub-sink \
    pubsub.googleapis.com/projects/PROJECT/topics/log-topic \
    --log-filter='resource.type="k8s_container" severity=ERROR'

# List sinks
gcloud logging sinks list

# Grant sink service account write access
gcloud projects add-iam-policy-binding PROJECT \
    --member="serviceAccount:SINK_SERVICE_ACCOUNT" \
    --role="roles/bigquery.dataEditor"

Log Buckets

# Create custom log bucket
gcloud logging buckets create my-bucket \
    --location=global \
    --retention-days=30 \
    --description="Application logs"

# Update retention
gcloud logging buckets update my-bucket \
    --location=global \
    --retention-days=90

# Create sink to custom bucket
gcloud logging sinks create app-logs-sink \
    logging.googleapis.com/projects/PROJECT/locations/global/buckets/my-bucket \
    --log-filter='resource.type="k8s_container"'

Log Router

# Exclusion filter (reduce costs)
gcloud logging sinks create _Default \
    --log-filter='NOT resource.type="k8s_container" OR severity>=WARNING'

# Create exclusion
gcloud logging sinks update _Default \
    --add-exclusion=name=exclude-debug,filter='severity=DEBUG'

Logs Explorer CLI

# Read logs
gcloud logging read 'resource.type="k8s_container"' \
    --limit=100 \
    --format=json

# Stream logs (tail)
gcloud logging tail 'resource.type="k8s_container" resource.labels.namespace_name="production"'

# Read with time range
gcloud logging read 'resource.type="k8s_container"' \
    --freshness=1h

# Write log entry
gcloud logging write my-log "Test log message" \
    --severity=INFO \
    --payload-type=text

Log-Based Alerts

# Create alert from log-based metric
gcloud alpha monitoring policies create \
    --policy-from-file=log-alert.json
// log-alert.json
{
  "displayName": "Error Rate Alert",
  "conditions": [{
    "displayName": "Error logs > 10/min",
    "conditionThreshold": {
      "filter": "metric.type=\"logging.googleapis.com/user/error-count\"",
      "comparison": "COMPARISON_GT",
      "thresholdValue": 10,
      "duration": "60s",
      "aggregations": [{
        "alignmentPeriod": "60s",
        "perSeriesAligner": "ALIGN_RATE"
      }]
    }
  }],
  "notificationChannels": ["projects/PROJECT/notificationChannels/CHANNEL_ID"]
}

Error Reporting

# Error Reporting automatically groups errors from logs
# Ensure your application logs include stack traces

# View errors
gcloud beta error-reporting events list

# Delete error group
gcloud beta error-reporting events delete

Common Commands

# List log names
gcloud logging logs list

# Delete logs (careful!)
gcloud logging logs delete LOG_NAME

# Get project logging settings
gcloud logging settings describe

# Update default retention
gcloud logging buckets update _Default \
    --location=global \
    --retention-days=30
