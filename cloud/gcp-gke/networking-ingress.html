GKE Ingress

# ingress.yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-ingress
  annotations:
    kubernetes.io/ingress.global-static-ip-name: "my-static-ip"
spec:
  ingressClassName: gce
  rules:
  - host: app.example.com
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: frontend-service
            port:
              number: 80
      - path: /api
        pathType: Prefix
        backend:
          service:
            name: api-service
            port:
              number: 80

Reserve Static IP

# Reserve global static IP
gcloud compute addresses create my-static-ip --global

# Get the IP address
gcloud compute addresses describe my-static-ip --global

SSL/TLS with Managed Certificates

# managed-cert.yaml
apiVersion: networking.gke.io/v1
kind: ManagedCertificate
metadata:
  name: my-cert
spec:
  domains:
    - app.example.com
---
# ingress with SSL
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-ingress
  annotations:
    kubernetes.io/ingress.global-static-ip-name: "my-static-ip"
    networking.gke.io/managed-certificates: "my-cert"
    kubernetes.io/ingress.class: "gce"
spec:
  rules:
  - host: app.example.com
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: my-service
            port:
              number: 80

HTTPS Redirect

apiVersion: networking.gke.io/v1beta1
kind: FrontendConfig
metadata:
  name: my-frontend-config
spec:
  redirectToHttps:
    enabled: true
    responseCodeName: MOVED_PERMANENTLY_DEFAULT
---
# Add to ingress annotations
annotations:
  networking.gke.io/v1beta1.FrontendConfig: "my-frontend-config"

Cloud Armor (WAF)

# Create security policy
gcloud compute security-policies create my-policy

# Add rules
gcloud compute security-policies rules create 1000 \
    --security-policy=my-policy \
    --expression="origin.region_code == 'CN'" \
    --action=deny-403

# Apply to backend
gcloud compute backend-services update BACKEND_NAME \
    --security-policy=my-policy
DNS Setup

After creating your Ingress, point your domain's DNS A record to the static IP address. SSL certificate provisioning takes 10-15 minutes.
