Cloud CDN

Cloud CDN caches content at Google's edge locations for faster delivery.

Enable CDN

# Enable on backend service
gcloud compute backend-services update my-backend \
    --enable-cdn \
    --global

GKE BackendConfig

apiVersion: cloud.google.com/v1
kind: BackendConfig
metadata:
  name: cdn-config
spec:
  cdn:
    enabled: true
    cachePolicy:
      includeHost: true
      includeProtocol: true
      includeQueryString: false

Cache Modes

ModeDescription
CACHE_ALL_STATICCache static content
USE_ORIGIN_HEADERSRespect Cache-Control
FORCE_CACHE_ALLCache everything

Cache Invalidation

# Invalidate path
gcloud compute url-maps invalidate-cdn-cache my-url-map \
    --path="/images/*" \
    --global

# Invalidate all
gcloud compute url-maps invalidate-cdn-cache my-url-map \
    --path="/*" \
    --global

TTL Settings

gcloud compute backend-services update my-backend \
    --enable-cdn \
    --default-ttl=3600 \
    --max-ttl=86400 \
    --global
