Cloud DNS

Cloud DNS is a managed DNS service for publishing domain names to the global DNS.

Create Managed Zone

# Public zone
gcloud dns managed-zones create my-zone \
    --dns-name="example.com." \
    --description="Public zone" \
    --visibility=public

# Private zone (for internal resolution)
gcloud dns managed-zones create internal-zone \
    --dns-name="internal.example.com." \
    --description="Private zone" \
    --visibility=private \
    --networks=my-vpc

DNS Records

# Add A record
gcloud dns record-sets create app.example.com. \
    --zone=my-zone \
    --type=A \
    --ttl=300 \
    --rrdatas="35.190.80.1"

# Add CNAME record
gcloud dns record-sets create www.example.com. \
    --zone=my-zone \
    --type=CNAME \
    --ttl=300 \
    --rrdatas="app.example.com."

# List records
gcloud dns record-sets list --zone=my-zone

GKE ExternalDNS

# Automatically manage DNS from Ingress
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-ingress
  annotations:
    external-dns.alpha.kubernetes.io/hostname: app.example.com
spec:
  rules:
  - host: app.example.com
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: my-service
            port:
              number: 80

Private DNS for GKE

# Create private zone for internal services
gcloud dns managed-zones create gke-internal \
    --dns-name="cluster.local." \
    --description="GKE internal DNS" \
    --visibility=private \
    --networks=my-vpc

# GKE services are automatically resolvable:
# service-name.namespace.svc.cluster.local

DNS Peering

# Peer DNS zones between VPCs
gcloud dns managed-zones create peer-zone \
    --dns-name="other.example.com." \
    --description="Peered zone" \
    --visibility=private \
    --networks=my-vpc \
    --target-network=other-vpc \
    --target-project=other-project

Common Commands

# List zones
gcloud dns managed-zones list

# Describe zone
gcloud dns managed-zones describe my-zone

# Delete record
gcloud dns record-sets delete app.example.com. \
    --zone=my-zone \
    --type=A

# Get nameservers (for domain registrar)
gcloud dns managed-zones describe my-zone \
    --format="value(nameServers)"
