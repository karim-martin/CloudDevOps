Cloud Storage

Cloud Storage is GCP's object storage service for storing and serving files, backups, static assets, and data.

Create Bucket

# Create bucket
gcloud storage buckets create gs://my-bucket \
    --location=us-central1 \
    --default-storage-class=STANDARD \
    --uniform-bucket-level-access

# With lifecycle and versioning
gcloud storage buckets create gs://my-bucket \
    --location=us \
    --default-storage-class=STANDARD \
    --uniform-bucket-level-access \
    --enable-versioning

Upload & Download

# Upload file
gcloud storage cp local-file.txt gs://my-bucket/

# Upload directory
gcloud storage cp -r ./local-dir gs://my-bucket/

# Download file
gcloud storage cp gs://my-bucket/file.txt ./

# Sync directory
gcloud storage rsync -r ./local-dir gs://my-bucket/backup/

gsutil Commands

# List buckets
gsutil ls

# List objects
gsutil ls gs://my-bucket/
gsutil ls -l gs://my-bucket/  # With details

# Copy
gsutil cp file.txt gs://my-bucket/
gsutil cp -r dir/ gs://my-bucket/

# Move/rename
gsutil mv gs://my-bucket/old.txt gs://my-bucket/new.txt

# Delete
gsutil rm gs://my-bucket/file.txt
gsutil rm -r gs://my-bucket/dir/

# Get object info
gsutil stat gs://my-bucket/file.txt

Access Control

# Make bucket public (not recommended for sensitive data)
gcloud storage buckets add-iam-policy-binding gs://my-bucket \
    --member=allUsers \
    --role=roles/storage.objectViewer

# Grant access to service account
gcloud storage buckets add-iam-policy-binding gs://my-bucket \
    --member=serviceAccount:my-sa@project.iam.gserviceaccount.com \
    --role=roles/storage.objectAdmin

# Remove public access
gcloud storage buckets remove-iam-policy-binding gs://my-bucket \
    --member=allUsers \
    --role=roles/storage.objectViewer

Signed URLs

# Generate signed URL (requires service account key)
gsutil signurl -d 1h key.json gs://my-bucket/file.txt

# Using gcloud (with impersonation)
gcloud storage sign-url gs://my-bucket/file.txt \
    --duration=1h \
    --impersonate-service-account=my-sa@project.iam.gserviceaccount.com

Lifecycle Management

// lifecycle.json
{
  "rule": [
    {
      "action": {"type": "Delete"},
      "condition": {"age": 365}
    },
    {
      "action": {"type": "SetStorageClass", "storageClass": "NEARLINE"},
      "condition": {"age": 30}
    },
    {
      "action": {"type": "Delete"},
      "condition": {"isLive": false, "numNewerVersions": 3}
    }
  ]
}
# Apply lifecycle rules
gcloud storage buckets update gs://my-bucket --lifecycle-file=lifecycle.json

CORS Configuration

// cors.json
[
  {
    "origin": ["https://example.com"],
    "method": ["GET", "PUT", "POST"],
    "responseHeader": ["Content-Type"],
    "maxAgeSeconds": 3600
  }
]
# Apply CORS
gcloud storage buckets update gs://my-bucket --cors-file=cors.json

SDK Examples

Node.js

const { Storage } = require('@google-cloud/storage');
const storage = new Storage();
const bucket = storage.bucket('my-bucket');

// Upload
await bucket.upload('local-file.txt', { destination: 'remote-file.txt' });

// Download
await bucket.file('remote-file.txt').download({ destination: 'local-file.txt' });

// List
const [files] = await bucket.getFiles();
files.forEach(file => console.log(file.name));

// Delete
await bucket.file('file.txt').delete();

// Generate signed URL
const [url] = await bucket.file('file.txt').getSignedUrl({
  action: 'read',
  expires: Date.now() + 60 * 60 * 1000 // 1 hour
});

Python

from google.cloud import storage

client = storage.Client()
bucket = client.bucket('my-bucket')

# Upload
blob = bucket.blob('remote-file.txt')
blob.upload_from_filename('local-file.txt')

# Download
blob = bucket.blob('remote-file.txt')
blob.download_to_filename('local-file.txt')

# List
blobs = bucket.list_blobs()
for blob in blobs:
    print(blob.name)

# Delete
bucket.blob('file.txt').delete()

# Signed URL
url = blob.generate_signed_url(expiration=3600)

Common Commands

# List buckets
gcloud storage buckets list

# Describe bucket
gcloud storage buckets describe gs://my-bucket

# Delete bucket (must be empty)
gcloud storage buckets delete gs://my-bucket

# Delete bucket and contents
gcloud storage rm -r gs://my-bucket

# Get bucket size
gsutil du -s gs://my-bucket/
Bucket Names are Global

Bucket names must be globally unique across all of GCP. Use project ID or domain name as prefix to ensure uniqueness.
