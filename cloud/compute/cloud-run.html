Cloud Run

Cloud Run is a fully managed serverless platform for running stateless containers. It automatically scales from zero to thousands of instances based on traffic.

Services vs Jobs

TypeUse CaseTrigger
ServicesWeb apps, APIs, webhooksHTTP requests
JobsBatch processing, migrationsManual, scheduled, or event

Deploy a Service

# Deploy from source (builds automatically)
gcloud run deploy my-service \
    --source=. \
    --region=us-central1 \
    --allow-unauthenticated

# Deploy from container image
gcloud run deploy my-service \
    --image=us-central1-docker.pkg.dev/PROJECT_ID/repo/image:tag \
    --region=us-central1 \
    --port=8080 \
    --memory=512Mi \
    --cpu=1 \
    --min-instances=0 \
    --max-instances=100 \
    --allow-unauthenticated

# Deploy with environment variables
gcloud run deploy my-service \
    --image=us-central1-docker.pkg.dev/PROJECT_ID/repo/image:tag \
    --region=us-central1 \
    --set-env-vars="DB_HOST=10.0.0.1,ENV=production" \
    --set-secrets="DB_PASSWORD=db-password:latest"

Service Configuration

# Update service settings
gcloud run services update my-service \
    --region=us-central1 \
    --memory=1Gi \
    --cpu=2 \
    --timeout=300 \
    --concurrency=80 \
    --min-instances=1 \
    --max-instances=50

# Set CPU allocation
gcloud run services update my-service \
    --region=us-central1 \
    --cpu-throttling  # CPU only during requests (default)
    # or
    --no-cpu-throttling  # CPU always allocated (for background work)

Traffic Management

# Deploy new revision without traffic
gcloud run deploy my-service \
    --image=new-image:v2 \
    --region=us-central1 \
    --no-traffic

# Split traffic between revisions
gcloud run services update-traffic my-service \
    --region=us-central1 \
    --to-revisions=my-service-00001=50,my-service-00002=50

# Gradual rollout
gcloud run services update-traffic my-service \
    --region=us-central1 \
    --to-revisions=my-service-00002=10  # 10% to new version

# Rollback to previous revision
gcloud run services update-traffic my-service \
    --region=us-central1 \
    --to-revisions=my-service-00001=100

VPC Connector (Private Resources)

# Create VPC connector
gcloud compute networks vpc-access connectors create my-connector \
    --region=us-central1 \
    --network=my-vpc \
    --range=10.8.0.0/28 \
    --min-instances=2 \
    --max-instances=10

# Deploy with VPC connector
gcloud run deploy my-service \
    --image=my-image \
    --region=us-central1 \
    --vpc-connector=my-connector \
    --vpc-egress=private-ranges-only  # or all-traffic

Connect to Cloud SQL

# Deploy with Cloud SQL connection
gcloud run deploy my-service \
    --image=my-image \
    --region=us-central1 \
    --add-cloudsql-instances=PROJECT_ID:us-central1:my-db \
    --set-env-vars="DB_HOST=/cloudsql/PROJECT_ID:us-central1:my-db"

# Connection string in app (Unix socket)
# PostgreSQL: Host=/cloudsql/PROJECT:REGION:INSTANCE
# MySQL: Server=/cloudsql/PROJECT:REGION:INSTANCE

Authentication

# Allow public access
gcloud run deploy my-service \
    --allow-unauthenticated

# Require authentication
gcloud run deploy my-service \
    --no-allow-unauthenticated

# Grant invoker access to specific user/SA
gcloud run services add-iam-policy-binding my-service \
    --region=us-central1 \
    --member="serviceAccount:my-sa@project.iam.gserviceaccount.com" \
    --role="roles/run.invoker"

# Call authenticated service
curl -H "Authorization: Bearer $(gcloud auth print-identity-token)" \
    https://my-service-xxx-uc.a.run.app

Cloud Run Jobs

# Create a job
gcloud run jobs create my-job \
    --image=us-central1-docker.pkg.dev/PROJECT_ID/repo/job-image:tag \
    --region=us-central1 \
    --tasks=10 \
    --parallelism=5 \
    --task-timeout=3600 \
    --max-retries=3

# Execute job
gcloud run jobs execute my-job --region=us-central1

# Execute with overrides
gcloud run jobs execute my-job \
    --region=us-central1 \
    --tasks=20 \
    --update-env-vars="BATCH_SIZE=1000"

# Schedule job (Cloud Scheduler)
gcloud scheduler jobs create http my-scheduled-job \
    --location=us-central1 \
    --schedule="0 2 * * *" \
    --uri="https://us-central1-run.googleapis.com/apis/run.googleapis.com/v1/namespaces/PROJECT_ID/jobs/my-job:run" \
    --http-method=POST \
    --oauth-service-account-email=scheduler-sa@PROJECT_ID.iam.gserviceaccount.com

Custom Domains

# Map custom domain
gcloud run domain-mappings create \
    --service=my-service \
    --region=us-central1 \
    --domain=api.example.com

# View DNS records to configure
gcloud run domain-mappings describe \
    --domain=api.example.com \
    --region=us-central1

Common Commands

# List services
gcloud run services list

# Describe service
gcloud run services describe my-service --region=us-central1

# View logs
gcloud run services logs read my-service --region=us-central1 --limit=100

# Delete service
gcloud run services delete my-service --region=us-central1

# List revisions
gcloud run revisions list --service=my-service --region=us-central1

YAML Configuration

# service.yaml
apiVersion: serving.knative.dev/v1
kind: Service
metadata:
  name: my-service
spec:
  template:
    metadata:
      annotations:
        autoscaling.knative.dev/minScale: "1"
        autoscaling.knative.dev/maxScale: "100"
        run.googleapis.com/vpc-access-connector: my-connector
        run.googleapis.com/cloudsql-instances: project:region:instance
    spec:
      containerConcurrency: 80
      timeoutSeconds: 300
      serviceAccountName: my-sa@project.iam.gserviceaccount.com
      containers:
      - image: us-central1-docker.pkg.dev/project/repo/image:tag
        ports:
        - containerPort: 8080
        resources:
          limits:
            cpu: "2"
            memory: 1Gi
        env:
        - name: ENV
          value: production
        - name: DB_PASSWORD
          valueFrom:
            secretKeyRef:
              name: db-password
              key: latest
# Deploy from YAML
gcloud run services replace service.yaml --region=us-central1
Cold Starts

Cloud Run instances scale to zero by default. Set --min-instances=1 to keep at least one instance warm and avoid cold start latency for critical services.

Stateless Requirement

Cloud Run containers are stateless. Don't store data in the container filesystem - use Cloud Storage, databases, or Memorystore for persistent data.
